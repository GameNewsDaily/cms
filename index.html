<html lang="zh-Hant">

<head>

  <meta charset="UTF-8">

  <title>Game News Daily</title>

  <link rel="stylesheet" href="style.css">

</head>

<body>

  <header>

    <h1>Game News Daily</h1>

  </header>

  

  <main>

    <section id="latest-posts">

      <h2>最新文章</h2>

      <div id="posts-container"></div>

    </section>

  </main>


  <footer>

    <p>&copy; 2025 Game News Daily</p>

  </footer>


  <script>

    async function loadPosts() {

      const response = await fetch("posts/");

      const text = await response.text();

      const parser = new DOMParser();

      const doc = parser.parseFromString(text, 'text/html');


      const links = Array.from(doc.querySelectorAll("a"))

        .filter(link => link.href.endsWith(".html"))

        .slice(0, 5);


      const container = document.getElementById("posts-container");


      for (let link of links) {

        const postUrl = link.getAttribute("href");

        const res = await fetch("posts/" + postUrl);

        const html = await res.text();

        const postDoc = parser.parseFromString(html, 'text/html');

        const title = postDoc.querySelector("h1").textContent;

        const img = postDoc.querySelector("img");

        const imgSrc = img ? img.getAttribute("src") : "assets/images/uploads/sample.png";

        const preview = postDoc.querySelector("p")?.textContent || "";


        const article = document.createElement("article");

        article.innerHTML = `

          <a href="posts/${postUrl}">

            <img src="${imgSrc}" alt="${title}">

            <h3>${title}</h3>

            <p>${preview}</p>

          </a>

        `;

        container.appendChild(article);

      }

    }


    loadPosts();

  </script>

</body>

</html>
